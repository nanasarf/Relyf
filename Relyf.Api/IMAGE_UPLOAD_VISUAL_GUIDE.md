## ?? Code Examples Side-by-Side

### React: File Input

```tsx
// ? RECOMMENDED
function FileUpload({ projectId }) {
  const handleChange = async (e) => {
    const file = e.target.files[0];
    
    const formData = new FormData();
    formData.append('file', file);
    formData.append('ownerType', 'Project');
    formData.append('ownerId', projectId);
    
    const res = await fetch('/api/images/upload', {
      method: 'POST',
      body: formData
    });
    
    const { url } = await res.json();
    // ? url = "/uploads/images/abc-123.jpg"
  };
  
  return <input type="file" onChange={handleChange} />;
}
```

### React: Canvas Upload

```tsx
// ? For cropping, filters, etc.
function CanvasUpload({ projectId }) {
  const uploadCanvas = async (canvas) => {
    const dataUrl = canvas.toDataURL('image/jpeg');
    
    const res = await fetch('/api/images', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        ownerType: 'Project',
        ownerId: projectId,
        source: dataUrl
      })
    });
    
    const { url } = await res.json();
    // ? url = "/uploads/images/xyz-789.jpg"
  };
  
  // ... canvas editing logic
}
```

### React: External URL

```tsx
// ? For linking external images
function ExternalImage({ projectId, imageUrl }) {
  const addImage = async () => {

        source: 'url',
        url: imageUrl
      })
    });
    
    const { imageId } = await res.json();
    // ? Image reference saved
  };
}
```

## ??? Database Schema

### Image Table

```sql
CREATE TABLE app.Image (
    ImageId       INT PRIMARY KEY IDENTITY(1,1),
    OwnerType     NVARCHAR(50) NOT NULL,  -- "Item" | "Idea" | "Project"
    OwnerId       INT NOT NULL,
    Source        NVARCHAR(50) NOT NULL,  -- "upload" | "url" | "cloudinary"
    Url           NVARCHAR(500) NOT NULL,
    AltText       NVARCHAR(200) NULL,
    CreatedAtUtc  DATETIME2 NOT NULL DEFAULT SYSUTCDATETIME()
);
```
